export class PhysicsEngine {
  constructor() {
    this.particles = [];
    this.gravityConstant = 1.0;
    this.coulombConstant = 5.0;
  }

  // Velocity Verlet integration step
  step(dt) {
    this.particles.forEach(p => p.integratePosition(dt));
    this.resolveGravitationalForces();
    this.particles.forEach(p => p.integrateVelocity(dt));
  }
}
